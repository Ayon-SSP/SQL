SELECT 
    column_name AS alias_name
    COUNT(*), SUM(), AVG(), MIN(), MAX(), LENGTH(), 
    ROUND(), UCASE(), LCASE(), MID(), NOW(), FORMAT()...
    DISTINCT column_name,
FROM table_name
    JOIN table_name ON table_name.column_name = table_name.column_name
    LEFT JOIN table_name ON table_name.column_name = table_name.column_name
    RIGHT JOIN table_name ON table_name.column_name = table_name.column_name
    FULL JOIN table_name ON table_name.column_name = table_name.column_name
    CROSS JOIN table_name
    INNER JOIN table_name ON table_name.column_name = table_name.column_name
    NATURAL JOIN table_name
    UNION
    UNION ALL
    INTERSECT
    EXCEPT
WHERE column_name > 10 
    AND column_name IS NOT NULL
    AND column_name IN (value1, value2, value3)
    OR column_name BETWEEN value1 AND value2
    OR column_name LIKE 'value%'
    OR column_name IN/NOT IN (SELECT column_name FROM table_name WHERE column_name = value)
    AND column_name ALL/ANY/SOME/EXISTS (SELECT column_name FROM table_name WHERE column_name = value)
    AND column_name = value

GROUP BY 
    column_name
HAVING
    SUM(column_name) > 10
ORDER BY 
    column_name DESC
    column_name ASC
OFFSET 10 ROWS
FETCH NEXT 10 ROWS ONLY;
